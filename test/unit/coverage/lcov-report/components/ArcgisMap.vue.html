<script>
import { loadModules } from 'esri-loader'
 
export default {
  name: 'ArcgisMap',
  props: {
    // 模式，1是编辑，2是展示。
    mode: {
      type: Number,
      default: 1
    },
    // 类型，point是点，polygon还是多边形，
    // 暂时就这俩
    type: {
      type: Array,
      // default: () => ['point', 'polygon']
      default: null
    },
    // 点位信息，数组格式，例如[123,321]
    point: {
      type: Array,
      default: null
    },
    // 多边行，二维数组，[[123,321],[123,321],[123,321]],第一个和最后一个元素值相同
    polygon: {
      type: Array,
      default: null
    }
 
  },
  data () {
    return {
      view: null,
      graphicsLayer: null,
      Graphic: null,
      pointSymbol: {
        type: 'simple-marker', // autocasts as new SimpleMarkerSymbol()
        color: [226, 119, 40],
        outline: {
          // autocasts as new SimpleLineSymbol()
          color: [255, 255, 255],
          width: 2
        }
      },
      polygonSymbol: {
        type: 'simple-fill', // autocasts as new SimpleFillSymbol()
        color: 'rgba(138,43,226, 0.8)',
        style: 'solid',
        outline: {
          color: 'white',
          width: 1
        }
      }
    }
  },
 
  mounted () {
    // lazy load the required ArcGIS API for JavaScript modules and CSS
    loadModules(['esri/Map',
      'esri/views/MapView',
      'esri/widgets/Sketch',
      'esri/layers/GraphicsLayer',
      'esri/layers/TileLayer',
 
      'esri/geometry/Point',
      'esri/Graphic'], { css: true })
      .then(([Map, MapView, Sketch, GraphicsLayer, TileLayer, Point, Graphic]) => {
        this.Graphic = Graphic
        var xyLayer = new TileLayer(
          {
            url: 'http://www.tianditucd.cn:8108/sswjs234basemapvector/arcgis/rest/services/BASEMAP/Vector/MapServer',
            // url: 'http://www.tianditucd.cn:8108/sswjs234basemapraster/arcgis/rest/services/BASEMAP/Raster/MapServer',
            // url: 'http://172.24.133.22:6080/arcgis/rest/services/BaseMap/OutSideVector_QSX/MapServer',
            id: 'base',
            opacity: 1
          }
        )
        //    var slLayer = new TileLayer(
        //   {
        //     // url: "http://www.tianditucd.cn:8108/sswjs234basemapvector/arcgis/rest/services/BASEMAP/Vector/MapServer",
        //     url: 'http://www.tianditucd.cn:8108/sswjs234basemaprasterlab/arcgis/rest/services/BASEMAP/RasterLab/MapServer',
        //     id: 'base',
        //     opacity: 0.75
        //   }
        // )
        this.graphicsLayer = new GraphicsLayer()
        const map = new Map({
          // basemap: 'topo-vector'
          layers: [xyLayer, this.graphicsLayer]
        })
        this.view = new MapView({
          container: 'viewDiv',
          map: map,
          zoom: 3,
          // center: [104.06107355178733, 30.661253624616165]
          center: {
            longitude: 104.06284532077233,
            latitude: 30.66057651892858,
            // x:
            spatialReference: 4490
          },
          ui: {
            components: ['zoom']
          }
        })
        // 加右上角功能框
        if (this.mode === 1) {
          const sketch =
              new Sketch({
                layer: this.graphicsLayer,
                view: this.view,
                // graphic will be selected as soon as it is created
                creationMode: 'update',
                // 多边形选择只支持点击，不支持拖动
                defaultCreateOptions: { mode: 'click' },
                availableCreateTools: this.type
                // availableCreateTools: ['point', 'polygon']
              })
 
          this.view.ui.add(sketch, 'top-right')
          sketch.on('create', (event) => {
            // console.log('create', event)
            // check if the create event's state has changed to complete indicating
            // the graphic create operation is completed.
 
            if (event.state === 'complete') {
              // this.graphicsLayer.removeAll()
              this.handleEventGraphic(event.graphic)
            }
          })
          sketch.on('update', (event) => {
            // console.log('update', event)
            // check if the create event's state has changed to complete indicating
            // the graphic create operation is completed.
            if (event.state === 'complete') {
              this.handleEventGraphic(event.graphics[0])
            }
          })
        }
        this.view.when(() => {
          // this.view.on('click', (event) => {
          //   this.view.hitTest(event).then((response) => {
          //     // console.log(response)
          //     if (response.results.length) {
          //       const graphic = response.results.filter((result) => {
          //         const layerId = result.graphic.layer.id
          //         return layerId === 'river'
          //       })[0].graphic
          //       if (graphic) {
          //         console.log('graphic.attributes', graphic.attributes)
          //         // const nodecode = graphic.attributes.REGIONCODE
          //       }
          //     }
          //   })
          // })
          if (this.point) {
            this.drawPoint()
          }
        })
        // function gotoTarget () {
        //   var inputStr = document.getElementById('gotoCoord').value
        //   console.log('点击进来', inputStr)
        //   if (!inputStr || inputStr.length === 0) {
        //     alert('未输入坐标点')
        //     return
        //   }
 
        //   var arr = inputStr.trim().split(',')
        //   if (arr.length !== 2) {
        //     alert('格式不正确')
        //     return
        //   }
        //   var pt = new Point({
        //     latitude: arr[1],
        //     longitude: arr[0],
        //     spatialReference: 4490
        //   })
        //   var markerSymbol = {
        //     type: 'simple-marker', // autocasts as new SimpleMarkerSymbol()
        //     color: [226, 119, 40],
        //     outline: {
        //       // autocasts as new SimpleLineSymbol()
        //       color: [255, 255, 255],
        //       width: 2
        //     }
        //   }
 
        //   // Create a graphic and add the geometry and symbol to it
        //   var pointGraphic = new Graphic({
        //     geometry: pt,
        //     symbol: markerSymbol
        //   })
        //   riverGraphicsLayer.add(pointGraphic)
        //   this.view.goTo({ target: pt, zoom: 10 })
        // }
        // this.handleEventGraphic(graphic)
      })
  },
  methods: {
    setPoint (data) {
      this.$emit('getPoint', data)
    },
    setPolygon (data) {
      this.$emit('getPolygon', data)
    },
    handleEventGraphic (graphic) {
      var type = graphic.geometry.type
      // 多边形,第一个和最后一个是一样的
      if (type === 'polygon') {
        var rings = graphic.geometry.rings
        // rings结构是三层数组，取第一个
        console.log('获取的多边形信息', JSON.stringify(rings[0]))
        this.setPolygon(rings[0])
        // alert('51区域数据已经打印在控制台')
      } else if (type === 'point') {
        // 点
        // const str = `lng:${graphic.geometry.longitude},lat:${graphic.geometry.latitude}`
        // console.log(str)
        // 另一种格式
        // 数组格式
        var parr = [graphic.geometry.longitude, graphic.geometry.latitude]
        console.log('获取的点信息', JSON.stringify(parr))
        this.setPoint(parr)
        // alert('51点位数据已经打印在控制台')
      }
      //  else if (type === 'polyline') {
      //   var paths = graphic.geometry.paths
      //   // 组装51格式的对象
      //   // { coord: '104.066724,30.700633', coord_z: '500.0' }
      //   const data = []
      //   const poiArray = paths[0]

      //   }
      //   var str = JSON.stringify(data)
      //   console.log('str', str)
      //   // alert('51区域数据已经打印在控制台')
      // }
    },
    drawPoint: function () {
      this.graphicsLayer.removeAll()
 
      // Create a graphic and add the geometry and symbol to it
      const pointGraphic = new this.Graphic({
        geometry: {
          type: 'point',
          longitude: this.point[0],
          latitude: this.point[1],
          spatialReference: 4490
        },
        symbol: this.pointSymbol
      })
      this.graphicsLayer.add(pointGraphic)
      this.view.goTo({ target: pointGraphic })
      console.log('传进来的点', this.point)
    },
    drawPolygon: function () {
      this.graphicsLayer.removeAll()
      // Create a graphic and add the geometry and symbol to it
      const polygonGraphic = new this.Graphic({
        geometry: {
          type: 'polygon',
          rings: this.polygon,
          spatialReference: 4490
        },
        symbol: this.polygonSymbol
      })
      this.graphicsLayer.add(polygonGraphic)
      this.view.goTo({ target: polygonGraphic })
      console.log('传进来的区域', this.polygon)
    }
 
  },
  watch: {
    point: {
      immediate: false,
      handler (newVal) {
        if (this.graphicsLayer && this.point) {
          this.drawPoint()
        }
      }
    },
    polygon: {
      immediate: false,
      handler (newVal) {
        if (this.graphicsLayer && this.polygon) {
          this.drawPolygon()
        }
      }
    }
  },
  beforeDestroy () {
    if (this.view) {
      // destroy the map view
      this.view.container = null
    }
  }
}
 
</script>
